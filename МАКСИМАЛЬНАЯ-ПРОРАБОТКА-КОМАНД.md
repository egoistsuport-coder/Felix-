## 1. 🌐 /translate - МАКСИМАЛЬНАЯ ВЕРСИЯ

### Базовая функция
```
/translate en Привет, как дела?
```

### Расширенные возможности

#### A. Автоопределение языка
```javascript
/translate en [текст]  // Явное указание
/translate auto [текст]  // Автоопределение исходного языка
```

#### B. Множественный перевод
```javascript
/translate en,es,de Привет!
// Переводит сразу на 3 языка
```

#### C. Режимы перевода
```javascript
/translate en:formal Привет!  // Формальный стиль
/translate en:casual Здравствуйте!  // Неформальный
/translate en:technical API документация  // Технический
/translate en:literary Красивый закат  // Литературный
```

#### D. С контекстом
```javascript
/translate en:context=business Договор подписан
// Учитывает бизнес-контекст
```

#### E. Обратный перевод (проверка качества)
```javascript
/translate en:verify Привет, как дела?
// Показывает: оригинал → перевод → обратный перевод
```

### Полная реализация

```javascript
async function translateTextPro(text, options = {}) {
  const {
    targetLang = 'en',
    sourceLang = 'auto',
    style = 'neutral',
    context = '',
    verify = false,
    multiple = false
  } = options;

  // Множественный перевод
  if (multiple) {
    const langs = targetLang.split(',');
    const translations = {};
    
    for (const lang of langs) {
      const result = await translateSingle(text, lang, style, context);
      translations[lang] = result;
    }
    
    return formatMultipleTranslations(translations);
  }

  // Обычный перевод
  const stylePrompts = {
    formal: 'Используй формальный, официальный стиль.',
    casual: 'Используй неформальный, разговорный стиль.',
    technical: 'Используй технический, профессиональный язык.',
    literary: 'Используй литературный, художественный стиль.',
    neutral: 'Используй нейтральный стиль.'
  };

  const contextPrompt = context ? `\nКонтекст: ${context}` : '';
  const sourcePrompt = sourceLang !== 'auto' 
    ? `Исходный язык: ${langNames[sourceLang]}.` 
    : 'Автоматически определи исходный язык.';

  const systemPrompt = `Ты профессиональный переводчик.
${sourcePrompt}
Переведи текст на ${langNames[targetLang]} язык.
${stylePrompts[style]}
${contextPrompt}
Сохрани смысл и нюансы оригинала.`;

  const translation = await groq.chat.completions.create({
    messages: [
      { role: 'system', content: systemPrompt },
      { role: 'user', content: text }
    ],
    model: 'llama-3.3-70b-versatile',
    temperature: 0.3,
    max_tokens: 1024
  });

  const result = translation.choices[0]?.message?.content;

  // Обратный перевод для проверки
  if (verify) {
    const backTranslation = await translateSingle(result, sourceLang, 'neutral');
    return formatVerifiedTranslation(text, result, backTranslation, targetLang);
  }

  return result;
}

function formatMultipleTranslations(translations) {
  let output = '🌐 <b>Переводы:</b>\n\n';
  for (const [lang, text] of Object.entries(translations)) {
    output += `<b>${langNames[lang]}:</b>\n${text}\n\n`;
  }
  return output;
}

function formatVerifiedTranslation(original, translation, backTranslation, lang) {
  return `🌐 <b>Перевод с проверкой:</b>

<b>Оригинал:</b>
${original}

<b>Перевод (${langNames[lang]}):</b>
${translation}

<b>Обратный перевод:</b>
${backTranslation}

<i>💡 Сравните оригинал и обратный перевод для оценки качества</i>`;
}
```

### Использование
```
/translate en Привет!
/translate en:formal Привет!
/translate en,es,de Привет!
/translate en:verify Привет!
/translate en:context=business Договор подписан
```

async function brainstormIdeasPro(topic, options = {}) {
  const {
    count = 10,
    categorize = false,
    rate = false,
    creativity = 'medium',
    detailed = false
  } = options;

  const creativityPrompts = {
    low: 'Генерируй практичные, реалистичные идеи.',

  };

  const structurePrompt = categorize 
    ? '\n\nГруппируй идеи по категориям.'
    : '';

    ? '\n\nДля каждой идеи добавь:\n- Сложность (1-5)\n- Потенциал (1-5)\n- Время реализации'
    : '';

  const detailPrompt = detailed 
    ? '\n\nДля каждой идеи добавь подробное описание (2-3 предложения).'
    : '';

  const systemPrompt = `Ты креативный генератор идей.
Создай ${count} интересных и практичных идей по теме: "${topic}".
${creativityPrompts[creativity]}${structurePrompt}${ratingPrompt}${detailPrompt}

Формат:
1. [Название идеи]
   [Описание]
   ${rate ? '[Сложность: X/5, Потенциал: X/5, Время: X]' : ''}`;

  const ideas = await groq.chat.completions.create({
    messages: [
      { role: 'system', content: systemPrompt },
      { role: 'user', content: topic }
    ],
    model: 'llama-3.3-70b-versatile',
    temperature: creativity === 'wild' ? 0.9 : creativity === 'high' ? 0.8 : 0.7,
    max_tokens: 1500
  });

  return ideas.choices[0]?.message?.content;
}
```

function getDetailedStats(userId, options = {}) {
  const {
    detailed = false,
    chart = false,
    commandsOnly = false,
    timeOnly = false
  } = options;

  const stats = userStats.get(userId) || {
    totalMessages: 0,
    commands: {},
    voiceMessages: 0,
    startTime: Date.now(),
    sessions: []
  };

  const sessionTime = Math.floor((Date.now() - stats.startTime) / 1000 / 60);
  const hours = Math.floor(sessionTime / 60);
  const minutes = sessionTime % 60;

  // Топ команд
  const topCommands = Object.entries(stats.commands)
    .sort((a, b) => b[1] - a[1])
    .slice(0, 10);

  // Базовая статистика
  let output = `📊 <b>Статистика использования</b>\n\n`;

  if (!commandsOnly && !timeOnly) {
    output += `💬 <b>Сообщения:</b> ${stats.totalMessages}\n`;
    output += `🎤 <b>Голосовых:</b> ${stats.voiceMessages}\n`;
    output += `⚡ <b>Команд:</b> ${Object.keys(stats.commands).length}\n`;
    output += `⏱ <b>Время сессии:</b> ${hours}ч ${minutes}мин\n\n`;
  }

  if (timeOnly) {
    output += `⏱ <b>Временная статистика:</b>\n`;
    output += `Начало сессии: ${new Date(stats.startTime).toLocaleString('ru')}\n`;
    output += `Длительность: ${hours}ч ${minutes}мин\n`;
    output += `Сообщений в минуту: ${(stats.totalMessages / sessionTime).toFixed(2)}\n\n`;
  }

  if (!timeOnly) {
    // Топ команд
    if (topCommands.length > 0) {
      output += `🏆 <b>Топ команд:</b>\n`;
      topCommands.forEach(([cmd, count], index) => {
        const bar = '█'.repeat(Math.ceil(count / topCommands[0][1] * 10));
        output += `${index + 1}. ${cmd}: ${count}x ${chart ? bar : ''}\n`;
      });
      output += `\n`;
    }

    // Любимая команда
    if (topCommands[0]) {
      output += `💎 <b>Любимая команда:</b> ${topCommands[0][0]} (${topCommands[0][1]}x)\n\n`;
    }
  }

  if (detailed) {
    // Детальная статистика
    output += `📈 <b>Детальная статистика:</b>\n`;
    output += `Текстовых сообщений: ${stats.totalMessages - stats.voiceMessages}\n`;
    output += `Голосовых сообщений: ${stats.voiceMessages}\n`;
    output += `Процент голосовых: ${((stats.voiceMessages / stats.totalMessages) * 100).toFixed(1)}%\n`;
    output += `Среднее сообщений/час: ${(stats.totalMessages / (sessionTime / 60)).toFixed(1)}\n\n`;
  }

  output += `<i>💡 Статистика сбрасывается при перезапуске бота</i>`;

  return output;
}
```

```javascript
const cache = new Map();

function getCached(key, ttl = 3600000) {
  const cached = cache.get(key);
  if (cached && Date.now() - cached.time < ttl) {
    return cached.value;
  }
  return null;
}

function setCache(key, value) {
  cache.set(key, { value, time: Date.now() });
}
```

### 2. История команд
```javascript
const commandHistory = new Map();

function addToHistory(userId, command, result) {
  if (!commandHistory.has(userId)) {
    commandHistory.set(userId, []);
  }
  const history = commandHistory.get(userId);
  history.push({
    command,
    result: result.substring(0, 100),
    timestamp: Date.now()
  });
  if (history.length > 50) {
    commandHistory.set(userId, history.slice(-50));
  }
}
```
