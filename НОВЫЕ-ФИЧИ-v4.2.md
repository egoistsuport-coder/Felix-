## 💻 Пример реализации

### /translate
```javascript
async function translateText(text, targetLang) {
  const langNames = {
    en: 'английский',
    ru: 'русский',
    es: 'испанский',
    de: 'немецкий',
    fr: 'французский'
  };

  const completion = await groq.chat.completions.create({
    messages: [
      {
        role: 'system',
        content: `Ты профессиональный переводчик. Переведи текст на ${langNames[targetLang]} язык. Сохрани смысл и стиль оригинала.`
      },
      { role: 'user', content: text }
    ],
    model: 'llama-3.3-70b-versatile',
    temperature: 0.3,
    max_tokens: 1024
  });

  return completion.choices[0]?.message?.content;
}

// В webhook.js
if (text.startsWith('/translate ')) {
  const parts = text.replace('/translate ', '').split(' ');
  const targetLang = parts[0];
  const textToTranslate = parts.slice(1).join(' ');
  
  if (!textToTranslate) {
    await sendMessage(chatId, 'Используйте: /translate <язык> <текст>\nПример: /translate en Привет!');
    return res.status(200).json({ ok: true });
  }
  
  await sendMessage(chatId, '🔄 Перевожу...');
  
  const translated = await translateText(textToTranslate, targetLang);
  await sendMessage(chatId, `🌐 <b>Перевод:</b>\n\n${translated}`);
  
  return res.status(200).json({ ok: true });
}
```

### /stats (с in-memory хранением)
```javascript
// В начале файла
const userStats = new Map();

// Функция для обновления статистики
function updateStats(userId, action) {
  if (!userStats.has(userId)) {
    userStats.set(userId, {
      totalMessages: 0,
      commands: {},
      voiceMessages: 0,
      startTime: Date.now()
    });
  }
  
  const stats = userStats.get(userId);
  stats.totalMessages++;
  
  if (action.startsWith('/')) {
    stats.commands[action] = (stats.commands[action] || 0) + 1;
  } else if (action === 'voice') {
    stats.voiceMessages++;
  }
}

// В webhook.js
if (text === '/stats') {
  const stats = userStats.get(userId) || {
    totalMessages: 0,
    commands: {},
    voiceMessages: 0,
    startTime: Date.now()
  };
  
  const sessionTime = Math.floor((Date.now() - stats.startTime) / 1000 / 60);
  const topCommand = Object.entries(stats.commands)
    .sort((a, b) => b[1] - a[1])[0];
  
  const statsText = `📊 <b>Ваша статистика:</b>

💬 Сообщений: ${stats.totalMessages}
🎤 Голосовых: ${stats.voiceMessages}
⚡ Команд использовано: ${Object.keys(stats.commands).length}
⏱ Время сессии: ${sessionTime} мин

${topCommand ? `🏆 Любимая команда: ${topCommand[0]} (${topCommand[1]}x)` : ''}

<i>Статистика сбрасывается при перезапуске бота</i>`;
  
  await sendMessage(chatId, statsText);
  return res.status(200).json({ ok: true });
}
```
